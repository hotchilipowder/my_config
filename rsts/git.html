  push:

      - name: Install

        run: |
          pip install -r requirements.txt
          sphinx-build -b html docs build

      - name: Make snippets to rst
        shell: bash
        working-directory: snippets
        run: |
          python3 snippets.py

      - name: Deploy Website
        uses: peaceiris/actions-gh-pages@v3
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
          publish_dir: ./build
